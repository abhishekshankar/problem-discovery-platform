# Signal pipeline — template crontab entries (PRD ops cadence)
# Install: ./scripts/install-cron.sh (from repo root)
# Replace REPO_PARENT with the directory that contains the `problem_discovery` package on PYTHONPATH
# Example: if code lives in ~/proj/src/problem_discovery, set REPO_PARENT=~/proj/src

SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin

# §8 extraction + §9 clustering (4-hourly)
0 */4 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal extract --limit 200 >> /tmp/signal-extract.log 2>&1
15 */4 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal cluster >> /tmp/signal-cluster.log 2>&1

# §14.4 mixed extractor versions (weekly Sun 05:00)
0 5 * * 0 cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal cluster-version-hygiene >> /tmp/signal-version.log 2>&1

# §15.3 stale briefs + §11.1 regen (daily)
0 2 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal archive-stale-briefs >> /tmp/signal-archive.log 2>&1
0 3 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal regen-stale-briefs >> /tmp/signal-regen.log 2>&1

# §7.6 collector health + §12.3 digest + §16.2 rebalance (daily)
0 9 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal collector-health >> /tmp/signal-health.log 2>&1
0 8 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal digest >> /tmp/signal-digest.log 2>&1
0 10 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal source-rebalance >> /tmp/signal-rebalance.log 2>&1

# §16.1 patterns + §16.4 exploration (daily)
0 11 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal propose-patterns >> /tmp/signal-patterns.log 2>&1
30 6 * * * cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal exploration-cluster >> /tmp/signal-explore.log 2>&1

# §15.2 calibration (weekly Mon)
30 5 * * 1 cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal calibrate >> /tmp/signal-calibrate.log 2>&1

# §9 hygiene (weekly Sun)
0 4 * * 0 cd REPO_PARENT && PYTHONPATH=. python -m problem_discovery.signal cluster-hygiene >> /tmp/signal-hygiene.log 2>&1
